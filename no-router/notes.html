Boilerplate

You repeat more code when every page includes the <head> tag. Routers reduce boilerplate.

<!doctype html>
<html>
  <head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1">
    <meta name="viewport" content="width=device-width, minimum-scale=1.0, initial-scale=1.0, user-scalable=yes">
    <title>no-router</title>
    <script src="/bower_components/platform/platform.js"></script>
    <link rel="stylesheet" href="/css/styles.css" shim-shadowdom>
    <link rel="import" href="/bower_components/polymer/polymer.html">
    <link rel="import" href="/layouts/sidebar-layout.html">
  </head>
  <body unresolved>
    <sidebar-layout selected="notes">
      <div class="title">Notes</div>

      <p>Hello World.</p>

    </sidebar-layout>
  </body>
</html>
